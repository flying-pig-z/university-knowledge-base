# 生产者消费者问题
## 实验要求
### 编写主函数逻辑框架
![](https://cdn.nlark.com/yuque/0/2024/png/42768076/1734406077389-95bc8600-1943-4b8f-a7c3-54f338ac90ca.png)

### 封装信号操作函数
【1】MySem newsem(int initVal)

创建新的信号量，初值为initVal，完成后返回信号量的ID（typedef int MySem;）

【2】void psem(MySem semID)

对ID为semID的信号量做p

【3】void vsem(MySem semID)

对ID为semID的信号量做v

【4】void freesem(MySem semID)

注销ID为semID的信号量

### 实现init函数
在init()中添加代码，请求用户输入仓库库存，然后申请并初始化共享存储区（仓库数组和变量in,out），以及各种信号量。

### 完善Pro()和Con()函数
在Pro()和Con()中添加代码，完成复杂PC问题（多生产者、多消费者）的模拟。 

其中Pro()和Con()开始时在屏幕显示 ：<进程类型（P或C）> <进程id> started. 

结束时

Pro()显示：P <进程id> put an item to <仓库位置>. 

Con()显示：C <进程id> got an item from <仓库位置>.

## 实验代码及原理
**实验原理：**

![](https://cdn.nlark.com/yuque/0/2024/png/42768076/1734406077315-6547c0a7-e467-41ff-a173-105a3c0785ab.png)

但是生产者和消费者在对仓库操作的时候会产生并发问题。具体如下：

【1】多个生成者同时往同一格生成

【2】多个消费者同时消费同一格

【3】在仓库空位不足的时候仍进行生产

【4】在仓库没有物品的时候仍进行消费

**实验代码：**

```plain
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <sys/types.h>
#include <time.h>
#include <string.h>

typedef int MySem; // 定义信号量类型

int buffer_size;

// 信号量工具函数
MySem newsem(int initVal) {
    // 创建新的信号量
    MySem semID = semget(IPC_PRIVATE, 1, IPC_CREAT | 0666); 
    // 创建信号量错误处理
    if (semID < 0) {
        perror("semget error"); 
        exit(1);
    }
    // 设置信号量初始值
    semctl(semID, 0, SETVAL, initVal); 
    return semID;
}

void psem(MySem semID) {
    // 执行P操作 -- 信号量减一
    struct sembuf op = {0, -1, 0}; 
    semop(semID, &op, 1); 
}

void vsem(MySem semID) {
    // 执行V操作 -- 信号量加一
    struct sembuf op = {0, 1, 0};
    semop(semID, &op, 1); 
}

void freesem(MySem semID) {
    // 释放信号量
    semctl(semID, 0, IPC_RMID); 
}

// 共享内存结构
struct Warehouse {
    int *buffer; // 缓冲区
    int in, out; // 生产和消费指针
};

// 全局信号量和共享内存ID
MySem empty, full, mutex_pro, mutex_con; // 生产者和消费者独立的互斥信号量
int shmid; // 共享内存ID
struct Warehouse *warehouse; // 共享内存指针

void init() {
    // 创建共享内存,并附加共享内存
    shmid = shmget(IPC_PRIVATE, sizeof(struct Warehouse) + sizeof(int) * buffer_size, IPC_CREAT | 0666); 
    warehouse = (struct Warehouse *)shmat(shmid, NULL, 0);
    
    // 初始化生产指针和消费指针
    warehouse->in = 0; 
    warehouse->out = 0;
    warehouse->buffer = (int *)(warehouse + 1); // 动态分配缓冲区内存

    // 初始化信号量
    empty = newsem(buffer_size); // 空槽数量
    full = newsem(0); // 已填充槽数量
    mutex_pro = newsem(1); // 生产者互斥信号量
    mutex_con = newsem(1); // 消费者互斥信号量

    printf("初始化完成。仓库缓冲区大小: %d\n", buffer_size);
}

void Pro() {
    int item = rand() % 100; // 生产一个随机物品
    psem(empty); // 等待有空槽
    psem(mutex_pro); // 进入生产者的临界区
    
    // 临界区：将物品放入缓冲区
    warehouse->buffer[warehouse->in] = item;
    printf("生产者 %d 将物品放入 %d.\n", getpid() , warehouse->in);
    warehouse->in = (warehouse->in + 1) % buffer_size; // 更新生产指针
    
    vsem(mutex_pro); // 退出生产者临界区
    vsem(full); // 增加已填充槽数量
}

void Con() {
    psem(full); // 等待有填充槽
    psem(mutex_con); // 进入消费者的临界区
    
    // 临界区：从缓冲区中取出物品
    int item = warehouse->buffer[warehouse->out];
    printf("消费者 %d 从 %d 取出物品.\n", getpid(), warehouse->out);
    warehouse->out = (warehouse->out + 1) % buffer_size; // 更新消费指针
    
    vsem(mutex_con); // 退出消费者临界区
    vsem(empty); // 增加空槽数量
}

int main() {

    // 用户输入缓冲区大小
    printf("请输入缓冲区大小: ");
    scanf("%d", &buffer_size);

    // 初始化共享内存和信号量
    init();

    srand(time(NULL)); // 设置随机数种子

    // 使用循环模拟生产者和消费者的操作
    while (1) {
        pid_t pid = fork(); // 创建子进程
        
        if (pid < 0) {
            perror("fork error"); // 错误处理
            exit(1);
        } else if (pid == 0) { // 子进程
            int type = rand() % 2; // 随机选择生产者或消费者
            if (type == 0) {
                printf("生产者 %d 启动.\n", getpid());
                Pro(); // 调用生产者函数
            } else {
                printf("消费者 %d 启动.\n", getpid());
                Con(); // 调用消费者函数
            }
            exit(0); // 退出子进程
        } else { // 父进程
            sleep(rand() % 3); // 随机睡眠
        }
    }

    // 清理（在这个循环的例子中不可达，但在实际实现中是必要的）
    freesem(empty);
    freesem(full);
    freesem(mutex_pro);
    freesem(mutex_con);
    shmctl(shmid, IPC_RMID, NULL); // 删除共享内存

    return 0;
}
```

这里设置了四个个信号量：empty full mutex_pro mutex_con。分别用来解决上面四种并发问题。

【1】empty: 用于表示缓冲区中空的槽位数量，初始化为缓冲区大小。防止在在仓库没有空位的时候仍进行生产。

【2】Full: 用来表示缓冲区填满的槽位，初始化为0。防止仓库在没有物品的时候进行消费。

【3】mutex_pro: 初始化为1，控制每个时刻只能有一个生产者进行生产。防止多个生产者同时往同一格生产物品和同时更改生产指针。

【4】mutex_con: 初始化为1，控制每个时刻只能有一个消费者进行生产。防止多个消费者同时往同一格消费物品和同时更改消费指针。

## 实验结果与总结
可以看到用户先输入仓库库存，然后程序开始申请并初始化共享存储区(仓库数组和变量in,out)，以及各种信号量。

初始化成功后程序开始了多生产者、多消费者的模拟，并且在生产者和消费者启动和消费的时候打印日志。

通过日志可以看到轮流进行生产者和消费者的相关生产和消费操作的时候不会产生上面提到并发问题。说明成功通过信号量解决了生产者和消费者进程同步过程中的并发问题。

# 模拟动态分页
```plain
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 数据结构 -- 页表项
// 这个项目中的页表用链表表示,页表项是链表的结点
// 系统内存是分页的，分为一页一页；进程本身也是分页的，也分为一页一页
typedef struct {
    int pNo;    // 页号
    int bNo;    // 块号
    bool P;     // 中断位,判断页面是否已经被加载到内存中
    bool A;     // 访问位,true表示最近被访问过
} Page;

// 数据结构 -- 简化的进程控制块(PCB) 
// 进程本身也是分页的,所以这里记录了页表的长度和页表头指针,以便访问到进程的页表
typedef struct {
    int pageCount;     // 页表长度(即进程页面数)
    Page* pHead;    // 页表头指针
} PCB;

// 数据结构 -- 页队列结点
// 系统维护一个页队列承载装入的程序，通过LRU算法维护队列，淘汰时便于通过优先级进行淘汰
typedef struct PQNode {
    Page* page;              // 页表项指针
    struct PQNode* pre;      // 向前指针
    struct PQNode* next;     // 向后指针
} PQNode;
PQNode* pqHead;     // 页队列头指针
PQNode* pqTail;     // 页队列尾指针

// 系统内存属性
int pSize;          // 系统内存中页(块)大小
int bCount;         // 系统内存中总页(块)数
bool* bmp;          // 位示图 -- 用来表示系统内存页中哪个页已经装入，哪个页为空


// 函数声明
void SysInit();
PCB* NewProc();
void Access(PCB* p);
int Relocate(PCB* p, int logAddr);
void LackPage(Page* page);
void NonLackPage(Page* page);
void InsertNode(PQNode* pqNode);
void RemoveNode(PQNode* pqNode);
void Cleanup(PCB* proc);

// 页队列操作 -- 将一个页队列结点插入队列头部
void InsertNode(PQNode* pqNode) {
    if (!pqNode) return;
    
    // 在队列头部插入
    if (!pqHead) {
        pqHead = pqTail = pqNode;
    } else {
        pqNode->next = pqHead;
        pqHead->pre = pqNode;
        pqHead = pqNode;
    }
}

// 将一个页队列结点移除
void RemoveNode(PQNode* pqNode) {
    if (!pqNode) return;
    
    // 处理头节点
    if (pqNode == pqHead) {
        pqHead = pqNode->next;
        if (pqHead) pqHead->pre = NULL;
    }
    
    // 处理尾节点
    if (pqNode == pqTail) {
        pqTail = pqNode->pre;
        if (pqTail) pqTail->next = NULL;
    }
    
    // 处理中间节点
    if (pqNode->pre) pqNode->pre->next = pqNode->next;
    if (pqNode->next) pqNode->next->pre = pqNode->pre;
}

// 系统初始化 -- 页大小 页块数 页队列 位示图
void SysInit() {
    // 请求输入系统页大小
    printf("请输入页面大小: ");
    scanf("%d", &pSize);

    // 请求输入系统页块数
    printf("请输入总块数: ");
    scanf("%d", &bCount);
    
    // 初始化页队列，系统页的位示图
    pqHead = NULL;
    pqTail = NULL;
    bmp = (bool*)calloc(bCount, sizeof(bool));
}

// 进程PCB初始化
PCB* NewProc() {
    // 请求输入进程大小
    int pageCount;
    printf("请输入进程的页面数: ");
    scanf("%d", &pageCount);
    
    // 创建PCB并通过进程大小初始化PCB的页表
    PCB* p = (PCB*)malloc(sizeof(PCB));
    p->pageCount = pageCount; // 初始化页表大小
    p->pHead = (Page*)malloc(pageCount * sizeof(Page)); // 分配页表内存并初始化页表头指针
    // 初始化页表项
    for (int i = 0; i < pageCount; i++) {
        p->pHead[i].pNo = i;   // 直接访问结构体成员
        p->pHead[i].bNo = -1;
        p->pHead[i].P = false;
        p->pHead[i].A = false;
    }
    
    // 返回PCB指针
    return p;
}

// 模拟访问
void Access(PCB* p) {
    if (!p) return;
    
    // 请求输入想要访问的逻辑地址
    int logAddr;
    printf("请输入要访问的逻辑地址: ");
    scanf("%d", &logAddr);
    
    // 重定位以得到物理地址
    int phyAddr = Relocate(p, logAddr);
    
    // 输出结果
    if (phyAddr < 0) {
        printf("错误：访问被拒绝 - 非法地址\n");
    } else {
        printf("物理地址: %d\n", phyAddr);
    }
}

// 重定位的实现
int Relocate(PCB* p, int logAddr) {
    if (!p || logAddr < 0) return -1;
    
    // 计算页号和偏移量
    int pageNo = logAddr / pSize;
    int offset = logAddr % pSize;
    
    // 检查边界
    if (pageNo >= p->pageCount) return -1;
    
    // 获取页表项
    Page* page = &p->pHead[pageNo];
    
    // 检查页面是否在内存中
    if (!page->P) {
        LackPage(page);
    } else {
        NonLackPage(page);
    }
    
    // 计算物理地址
    return (page->bNo * pSize) + offset;
}

// 模拟缺页中断处理 -- 实现页面调度算法 -- LRU
void LackPage(Page* page) {
    if (!page) return;
    
    // 查找空闲块或牺牲页
    int blockNo = -1;
    
    // 寻找空闲块
    for (int i = 0; i < bCount; i++) {
        if (!bmp[i]) {
            blockNo = i;
            bmp[i] = true;
            break;
        }
    }
    
    // 如果没有空闲块，使用LRU算法选择牺牲页
    if (blockNo == -1 && pqTail) {
        Page* victimPage = pqTail->page;
        blockNo = victimPage->bNo;
        victimPage->P = false;
        victimPage->bNo = -1;
        RemoveNode(pqTail);
    }
    
    // 将块分配给页面
    if (blockNo != -1) {
        page->bNo = blockNo;
        page->P = true;
        page->A = true;
        
        // 创建新节点并插入队列头部
        PQNode* newNode = (PQNode*)malloc(sizeof(PQNode));
        newNode->page = page;
        newNode->pre = NULL;
        newNode->next = NULL;
        InsertNode(newNode);
    }
}

// 没有发生缺页中断 -- 将访问到页(page)的访问位置1
void NonLackPage(Page* page) {
    if (!page) return;
    
    // 设置访问位
    page->A = true;
    
    // 查找并将节点移动到队列头部
    PQNode* current = pqHead;
    while (current) {
        if (current->page == page) {
            RemoveNode(current);
            InsertNode(current);
            break;
        }
        current = current->next;
    }
}

// 清理资源
void Cleanup(PCB* proc) {
    if (proc) {
        // 释放页表
        if (proc->pHead) {
            free(proc->pHead);
        }
        free(proc);
    }
    
    // 清理页队列
    PQNode* current = pqHead;
    while (current) {
        PQNode* next = current->next;
        free(current);
        current = next;
    }
    
    // 释放位示图
    if (bmp) {
        free(bmp);
    }
}

int main() {
    // 初始化系统
    SysInit();
    
    // 创建进程
    PCB* proc = NewProc();
    
    // 模拟内存访问
    char continue_sim;
    do {
        Access(proc);
        printf("是否继续模拟？(y/n): ");
        scanf(" %c", &continue_sim);
    } while (continue_sim == 'y' || continue_sim == 'Y');
    
    // 清理资源
    Cleanup(proc);
    
    return 0;
}
```
