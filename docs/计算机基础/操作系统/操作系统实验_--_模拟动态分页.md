#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 数据结构 -- 页表项
// 这个项目中的页表用链表表示,页表项是链表的结点
// 系统内存是分页的，分为一页一页；进程本身也是分页的，也分为一页一页
typedef struct {
    int pNo;    // 页号
    int bNo;    // 块号
    bool P;     // 中断位,判断页面是否已经被加载到内存中
    bool A;     // 访问位,true表示最近被访问过
} Page;

// 数据结构 -- 简化的进程控制块(PCB) 
// 进程本身也是分页的,所以这里记录了页表的长度和页表头指针,以便访问到进程的页表
typedef struct {
    int pageCount;     // 页表长度(即进程页面数)
    Page* pHead;    // 页表头指针
} PCB;

// 数据结构 -- 页队列结点
// 系统维护一个页队列承载装入的程序，通过LRU算法维护队列，淘汰时便于通过优先级进行淘汰
typedef struct PQNode {
    Page* page;              // 页表项指针
    struct PQNode* pre;      // 向前指针
    struct PQNode* next;     // 向后指针
} PQNode;
PQNode* pqHead;     // 页队列头指针
PQNode* pqTail;     // 页队列尾指针

// 系统内存属性
int pSize;          // 系统内存中页(块)大小
int bCount;         // 系统内存中总页(块)数
bool* bmp;          // 位示图 -- 用来表示系统内存页中哪个页已经装入，哪个页为空


// 函数声明
void SysInit();
PCB* NewProc();
void Access(PCB* p);
int Relocate(PCB* p, int logAddr);
void LackPage(Page* page);
void NonLackPage(Page* page);
void InsertNode(PQNode* pqNode);
void RemoveNode(PQNode* pqNode);
void Cleanup(PCB* proc);

// 页队列操作 -- 将一个页队列结点插入队列头部
void InsertNode(PQNode* pqNode) {
    if (!pqNode) return;
    
    // 在队列头部插入
    if (!pqHead) {
        pqHead = pqTail = pqNode;
    } else {
        pqNode->next = pqHead;
        pqHead->pre = pqNode;
        pqHead = pqNode;
    }
}

// 将一个页队列结点移除
void RemoveNode(PQNode* pqNode) {
    if (!pqNode) return;
    
    // 处理头节点
    if (pqNode == pqHead) {
        pqHead = pqNode->next;
        if (pqHead) pqHead->pre = NULL;
    }
    
    // 处理尾节点
    if (pqNode == pqTail) {
        pqTail = pqNode->pre;
        if (pqTail) pqTail->next = NULL;
    }
    
    // 处理中间节点
    if (pqNode->pre) pqNode->pre->next = pqNode->next;
    if (pqNode->next) pqNode->next->pre = pqNode->pre;
}

// 系统初始化 -- 页大小 页块数 页队列 位示图
void SysInit() {
    // 请求输入系统页大小
    printf("请输入页面大小: ");
    scanf("%d", &pSize);

    // 请求输入系统页块数
    printf("请输入总块数: ");
    scanf("%d", &bCount);
    
    // 初始化页队列，系统页的位示图
    pqHead = NULL;
    pqTail = NULL;
    bmp = (bool*)calloc(bCount, sizeof(bool));
}

// 进程PCB初始化
PCB* NewProc() {
    // 请求输入进程大小
    int pageCount;
    printf("请输入进程的页面数: ");
    scanf("%d", &pageCount);
    
    // 创建PCB并通过进程大小初始化PCB的页表
    PCB* p = (PCB*)malloc(sizeof(PCB));
    p->pageCount = pageCount; // 初始化页表大小
    p->pHead = (Page*)malloc(pageCount * sizeof(Page)); // 分配页表内存并初始化页表头指针
    // 初始化页表项
    for (int i = 0; i < pageCount; i++) {
        p->pHead[i].pNo = i;   // 直接访问结构体成员
        p->pHead[i].bNo = -1;
        p->pHead[i].P = false;
        p->pHead[i].A = false;
    }
    
    // 返回PCB指针
    return p;
}

// 模拟访问
void Access(PCB* p) {
    if (!p) return;
    
    // 请求输入想要访问的逻辑地址
    int logAddr;
    printf("请输入要访问的逻辑地址: ");
    scanf("%d", &logAddr);
    
    // 重定位以得到物理地址
    int phyAddr = Relocate(p, logAddr);
    
    // 输出结果
    if (phyAddr < 0) {
        printf("错误：访问被拒绝 - 非法地址\n");
    } else {
        printf("物理地址: %d\n", phyAddr);
    }
}

// 重定位的实现
int Relocate(PCB* p, int logAddr) {
    if (!p || logAddr < 0) return -1;
    
    // 计算页号和偏移量
    int pageNo = logAddr / pSize;
    int offset = logAddr % pSize;
    
    // 检查边界
    if (pageNo >= p->pageCount) return -1;
    
    // 获取页表项
    Page* page = &p->pHead[pageNo];
    
    // 检查页面是否在内存中
    if (!page->P) {
        LackPage(page);
    } else {
        NonLackPage(page);
    }
    
    // 计算物理地址
    return (page->bNo * pSize) + offset;
}

// 模拟缺页中断处理 -- 实现页面调度算法 -- LRU
void LackPage(Page* page) {
    if (!page) return;
    
    // 查找空闲块或牺牲页
    int blockNo = -1;
    
    // 寻找空闲块
    for (int i = 0; i < bCount; i++) {
        if (!bmp[i]) {
            blockNo = i;
            bmp[i] = true;
            break;
        }
    }
    
    // 如果没有空闲块，使用LRU算法选择牺牲页
    if (blockNo == -1 && pqTail) {
        Page* victimPage = pqTail->page;
        blockNo = victimPage->bNo;
        victimPage->P = false;
        victimPage->bNo = -1;
        RemoveNode(pqTail);
    }
    
    // 将块分配给页面
    if (blockNo != -1) {
        page->bNo = blockNo;
        page->P = true;
        page->A = true;
        
        // 创建新节点并插入队列头部
        PQNode* newNode = (PQNode*)malloc(sizeof(PQNode));
        newNode->page = page;
        newNode->pre = NULL;
        newNode->next = NULL;
        InsertNode(newNode);
    }
}

// 没有发生缺页中断 -- 将访问到页(page)的访问位置1
void NonLackPage(Page* page) {
    if (!page) return;
    
    // 设置访问位
    page->A = true;
    
    // 查找并将节点移动到队列头部
    PQNode* current = pqHead;
    while (current) {
        if (current->page == page) {
            RemoveNode(current);
            InsertNode(current);
            break;
        }
        current = current->next;
    }
}

// 清理资源
void Cleanup(PCB* proc) {
    if (proc) {
        // 释放页表
        if (proc->pHead) {
            free(proc->pHead);
        }
        free(proc);
    }
    
    // 清理页队列
    PQNode* current = pqHead;
    while (current) {
        PQNode* next = current->next;
        free(current);
        current = next;
    }
    
    // 释放位示图
    if (bmp) {
        free(bmp);
    }
}

int main() {
    // 初始化系统
    SysInit();
    
    // 创建进程
    PCB* proc = NewProc();
    
    // 模拟内存访问
    char continue_sim;
    do {
        Access(proc);
        printf("是否继续模拟？(y/n): ");
        scanf(" %c", &continue_sim);
    } while (continue_sim == 'y' || continue_sim == 'Y');
    
    // 清理资源
    Cleanup(proc);
    
    return 0;
}
