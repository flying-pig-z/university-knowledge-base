### 2.封装信号操作函数
【1】MySem newsem(int initVal)

创建新的信号量，初值为initVal，完成后返回信号量的ID（typedef int MySem;）

【2】void psem(MySem semID)

对ID为semID的信号量做p

【3】void vsem(MySem semID)

对ID为semID的信号量做v

【4】void freesem(MySem semID)

注销ID为semID的信号量

### 3.实现init函数
在init()中添加代码，请求用户输入仓库库存，然后申请并初始化共享存储区（仓库数组和变量in,out），以及各种信号量。

### 4.完善Pro()和Con()函数
在Pro()和Con()中添加代码，完成复杂PC问题（多生产者、多消费者）的模拟。 

其中Pro()和Con()开始时在屏幕显示 ：<进程类型（P或C）> <进程id> started. 

结束时

Pro()显示：P <进程id> put an item to <仓库位置>. 

Con()显示：C <进程id> got an item from <仓库位置>.

## 二.实验代码及原理
**实验原理：**

![](https://cdn.nlark.com/yuque/0/2024/png/42768076/1734406077315-6547c0a7-e467-41ff-a173-105a3c0785ab.png)

但是生产者和消费者在对仓库操作的时候会产生并发问题。具体如下：

【1】多个生成者同时往同一格生成

【2】多个消费者同时消费同一格

【3】在仓库空位不足的时候仍进行生产

【4】在仓库没有物品的时候仍进行消费

**实验代码：**

```plain
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <sys/types.h>
#include <time.h>
#include <string.h>

typedef int MySem; // 定义信号量类型

int buffer_size;

// 信号量工具函数
MySem newsem(int initVal) {
    // 创建新的信号量
    MySem semID = semget(IPC_PRIVATE, 1, IPC_CREAT | 0666); 
    // 创建信号量错误处理
    if (semID < 0) {
        perror("semget error"); 
        exit(1);
    }
    // 设置信号量初始值
    semctl(semID, 0, SETVAL, initVal); 
    return semID;
}

void psem(MySem semID) {
    // 执行P操作 -- 信号量减一
    struct sembuf op = {0, -1, 0}; 
    semop(semID, &op, 1); 
}

void vsem(MySem semID) {
    // 执行V操作 -- 信号量加一
    struct sembuf op = {0, 1, 0};
    semop(semID, &op, 1); 
}

void freesem(MySem semID) {
    // 释放信号量
    semctl(semID, 0, IPC_RMID); 
}

// 共享内存结构
struct Warehouse {
    int *buffer; // 缓冲区
    int in, out; // 生产和消费指针
};

// 全局信号量和共享内存ID
MySem empty, full, mutex_pro, mutex_con; // 生产者和消费者独立的互斥信号量
int shmid; // 共享内存ID
struct Warehouse *warehouse; // 共享内存指针

void init() {
    // 创建共享内存,并附加共享内存
    shmid = shmget(IPC_PRIVATE, sizeof(struct Warehouse) + sizeof(int) * buffer_size, IPC_CREAT | 0666); 
    warehouse = (struct Warehouse *)shmat(shmid, NULL, 0);
    
    // 初始化生产指针和消费指针
    warehouse->in = 0; 
    warehouse->out = 0;
    warehouse->buffer = (int *)(warehouse + 1); // 动态分配缓冲区内存

    // 初始化信号量
    empty = newsem(buffer_size); // 空槽数量
    full = newsem(0); // 已填充槽数量
    mutex_pro = newsem(1); // 生产者互斥信号量
    mutex_con = newsem(1); // 消费者互斥信号量

    printf("初始化完成。仓库缓冲区大小: %d\n", buffer_size);
}

void Pro() {
    int item = rand() % 100; // 生产一个随机物品
    psem(empty); // 等待有空槽
    psem(mutex_pro); // 进入生产者的临界区
    
    // 临界区：将物品放入缓冲区
    warehouse->buffer[warehouse->in] = item;
    printf("生产者 %d 将物品放入 %d.\n", getpid() , warehouse->in);
    warehouse->in = (warehouse->in + 1) % buffer_size; // 更新生产指针
    
    vsem(mutex_pro); // 退出生产者临界区
    vsem(full); // 增加已填充槽数量
}

void Con() {
    psem(full); // 等待有填充槽
    psem(mutex_con); // 进入消费者的临界区
    
    // 临界区：从缓冲区中取出物品
    int item = warehouse->buffer[warehouse->out];
    printf("消费者 %d 从 %d 取出物品.\n", getpid(), warehouse->out);
    warehouse->out = (warehouse->out + 1) % buffer_size; // 更新消费指针
    
    vsem(mutex_con); // 退出消费者临界区
    vsem(empty); // 增加空槽数量
}

int main() {

    // 用户输入缓冲区大小
    printf("请输入缓冲区大小: ");
    scanf("%d", &buffer_size);

    // 初始化共享内存和信号量
    init();

    srand(time(NULL)); // 设置随机数种子

    // 使用循环模拟生产者和消费者的操作
    while (1) {
        pid_t pid = fork(); // 创建子进程
        
        if (pid < 0) {
            perror("fork error"); // 错误处理
            exit(1);
        } else if (pid == 0) { // 子进程
            int type = rand() % 2; // 随机选择生产者或消费者
            if (type == 0) {
                printf("生产者 %d 启动.\n", getpid());
                Pro(); // 调用生产者函数
            } else {
                printf("消费者 %d 启动.\n", getpid());
                Con(); // 调用消费者函数
            }
            exit(0); // 退出子进程
        } else { // 父进程
            sleep(rand() % 3); // 随机睡眠
        }
    }

    // 清理（在这个循环的例子中不可达，但在实际实现中是必要的）
    freesem(empty);
    freesem(full);
    freesem(mutex_pro);
    freesem(mutex_con);
    shmctl(shmid, IPC_RMID, NULL); // 删除共享内存

    return 0;
}
```

这里设置了四个个信号量：empty full mutex_pro mutex_con。分别用来解决上面四种并发问题。

【1】empty: 用于表示缓冲区中空的槽位数量，初始化为缓冲区大小。防止在在仓库没有空位的时候仍进行生产。

【2】Full: 用来表示缓冲区填满的槽位，初始化为0。防止仓库在没有物品的时候进行消费。

【3】mutex_pro: 初始化为1，控制每个时刻只能有一个生产者进行生产。防止多个生产者同时往同一格生产物品和同时更改生产指针。

【4】mutex_con: 初始化为1，控制每个时刻只能有一个消费者进行生产。防止多个消费者同时往同一格消费物品和同时更改消费指针。

## 三.实验结果与总结
可以看到用户先输入仓库库存，然后程序开始申请并初始化共享存储区(仓库数组和变量in,out)，以及各种信号量。

初始化成功后程序开始了多生产者、多消费者的模拟，并且在生产者和消费者启动和消费的时候打印日志。

通过日志可以看到轮流进行生产者和消费者的相关生产和消费操作的时候不会产生上面提到并发问题。说明成功通过信号量解决了生产者和消费者进程同步过程中的并发问题。

## 四.实验总结
这次实验主要是模拟了进程同步中非常出名的生产者-消费者问题。

并且我们通过设置信号量以及对信号量的pv操作解决生成者和消费者在进程同步中解决的并发问题。
